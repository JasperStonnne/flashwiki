  ## 目标

  将占位的 AdminGroupsPage 实现为完整的权限组管理页面。
  Manager 从 AdminShell 侧栏"权限组"进入 /admin/groups，
  能看到组列表（卡片）、创建/重命名/换组长/删除组、
  展开查看成员列表、添加/移除成员。

  ## 已有基础设施（直接使用，不要修改）

  ### API 端点（`api/endpoints.ts`）

  ```ts
  adminGroups: '/api/admin/groups'                                    // GET → GroupListItem[]
                                                                      // POST { name, leader_id } → GroupResponse
  adminGroup: (id: string) => `/api/admin/groups/${id}`               // PATCH { name?, leader_id? } → GroupResponse
                                                                      // DELETE → null
  adminGroupMembers: (id: string) => `/api/admin/groups/${id}/members`
                                                                      // GET → GroupMemberResponse[]
                                                                      // POST { user_id } → null
  adminGroupMember: (groupId: string, uid: string) =>
      `/api/admin/groups/${groupId}/members/${uid}`                    // DELETE → null
  adminUsers: '/api/admin/users'                                      // GET → UserResponse[]
  ```

  ### 类型（`api/types.ts`）

  ```ts
  interface GroupListItem {
    id: string; name: string; leader: GroupLeader;
    member_count: number; created_at: string
  }
  interface GroupLeader { id: string; display_name: string; email: string }
  interface GroupResponse { id: string; name: string; leader_id: string; created_at: string; updated_at: string }
  interface CreateGroupRequest { name: string; leader_id: string }
  interface UpdateGroupRequest { name?: string; leader_id?: string }
  interface GroupMemberResponse { group_id: string; user_id: string; joined_at: string }
  interface AddMemberRequest { user_id: string }
  interface UserResponse { id: string; email: string; display_name: string; role: UserRole; locale: string; created_at: string }
  ```

  #### 辅助函数（文件顶部，export function 之前）

  ```ts
  // 头像色板（与 AdminUsersPage 相同）
  const AVATAR_PALETTE = [
    'var(--primary)', 'var(--green)', 'var(--orange)',
    'var(--purple)', 'var(--rose)', 'var(--primary-dark)',
  ]

  function avatarColor(id: string): string {
    let hash = 0
    for (const char of id) { hash += char.charCodeAt(0) }
    return AVATAR_PALETTE[hash % AVATAR_PALETTE.length]
  }

  function firstChar(value: string): string {
    const trimmed = value.trim()
    if (trimmed === '') return '?'
    return Array.from(trimmed)[0] ?? '?'
  }

  function formatDate(iso: string): string {
    return iso.slice(0, 10)
  }

  function errorMessage(error: unknown): string {
    if (error instanceof Error && error.message.trim() !== '') return error.message
    return '请求失败'
  }
  ```

  #### 顶层组件 `AdminGroupsPage`

  ```
  状态:
  - groups: GroupListItem[]            — 组列表
  - allUsers: UserResponse[]           — 全部用户（用于组长下拉 + 添加成员下拉）
  - loading: boolean                   — 初始加载中
  - loadError: string | null           — 加载错误
  - reloadKey: number                  — 重试 / 创建后触发重新拉取
  - showCreateDialog: boolean          — 新建弹窗是否打开

  初始化 useEffect（依赖 [reloadKey]）:
  - let mounted = true
  - setLoading(true), setLoadError(null)
  - try:
      const [groupsData, usersData] = await Promise.all([
        request<GroupListItem[]>('GET', API_ENDPOINTS.adminGroups),
        request<UserResponse[]>('GET', API_ENDPOINTS.adminUsers),
      ])
      if (!mounted) return
      setGroups(groupsData)
      setAllUsers(usersData)
  - catch: if (mounted) setLoadError(errorMessage(error))
  - finally: if (mounted) setLoading(false)
  - cleanup: mounted = false

  回调:
  handleGroupUpdated(updated: GroupListItem):
    setGroups(prev => prev.map(g => g.id === updated.id ? updated : g))

  handleGroupDeleted(groupId: string):
    setGroups(prev => prev.filter(g => g.id !== groupId))

  handleCreated():
    setShowCreateDialog(false)
    setReloadKey(k => k + 1)   // 创建成功后重新拉列表（因为 POST 返回的 GroupResponse 不含 leader 详情）

  retry():
    setReloadKey(k => k + 1)

  渲染:
  <div className="admin-groups">
    <div className="admin-groups-header">
      <h1 className="admin-page-title">权限组管理</h1>
      <button className="ag-btn-primary" type="button"
              onClick={() => setShowCreateDialog(true)}>
        + 新建权限组
      </button>
    </div>

    {loading && <div className="admin-empty">加载中...</div>}

    {!loading && loadError && (
      <div className="admin-error">
        <p>{loadError}</p>
        <button className="admin-btn-secondary" type="button" onClick={retry}>重试</button>
      </div>
    )}

    {!loading && !loadError && groups.length === 0 && (
      <div className="admin-empty">暂无权限组</div>
    )}

    {!loading && !loadError && groups.length > 0 && (
      <div className="ag-groups-list">
        {groups.map(group => (
          <GroupCard
            key={group.id}
            group={group}
            allUsers={allUsers}
            onUpdated={handleGroupUpdated}
            onDeleted={handleGroupDeleted}
          />
        ))}
      </div>
    )}

    {showCreateDialog && createPortal(
      <CreateGroupDialog
        allUsers={allUsers}
        onClose={() => setShowCreateDialog(false)}
        onCreated={handleCreated}
      />,
      document.body,
    )}
  </div>
  ```

  #### 子组件 `GroupCard`

  ```
  Props:
  interface GroupCardProps {
    group: GroupListItem
    allUsers: UserResponse[]
    onUpdated: (group: GroupListItem) => void
    onDeleted: (groupId: string) => void
  }

  状态:
  - mode: 'view' | 'renaming' | 'changing-leader' | 'confirming-delete'
                                     — 卡片当前交互模式，同一时间只能有一种
  - editName: string                 — 重命名时的输入值
  - editLeaderId: string             — 换组长时选中的用户 ID
  - busy: boolean                    — 卡片级请求中（重命名/换组长/删除共用）
  - cardError: string | null         — 卡片级错误
  - expanded: boolean                — 成员区是否展开
  - members: GroupMemberResponse[] | null  — 成员列表，null 表示还没加载过
  - membersLoading: boolean          — 成员加载中
  - membersError: string | null      — 成员加载错误

  模式切换函数（每次切换时清除上一次的错误）:

  startRename():
    setMode('renaming')
    setEditName(group.name)
    setCardError(null)

  startChangeLeader():
    setMode('changing-leader')
    setEditLeaderId(group.leader.id)   // 预选当前组长
    setCardError(null)

  startDelete():
    setMode('confirming-delete')
    setCardError(null)

  cancelAction():
    setMode('view')
    setCardError(null)

  handleRename():
  - 校验: editName.trim() 非空且 ≤ 50 字符，否则 setCardError 并 return
  - setBusy(true), setCardError(null)
  - try:
      await request<GroupResponse>('PATCH', API_ENDPOINTS.adminGroup(group.id),
        { name: editName.trim() } satisfies UpdateGroupRequest)
      onUpdated({ ...group, name: editName.trim() })
      setMode('view')
  - catch: setCardError(errorMessage(error))
  - finally: setBusy(false)

  handleChangeLeader():
  - 校验: editLeaderId 非空，否则 return
  - setBusy(true), setCardError(null)
  - try:
      await request<GroupResponse>('PATCH', API_ENDPOINTS.adminGroup(group.id),
        { leader_id: editLeaderId } satisfies UpdateGroupRequest)
      // 从 allUsers 查新组长信息，构造更新后的 GroupListItem
      const newLeaderUser = allUsers.find(u => u.id === editLeaderId)
      const updatedGroup: GroupListItem = {
        ...group,
        leader: {
          id: editLeaderId,
          display_name: newLeaderUser?.display_name ?? editLeaderId.slice(0, 8),
          email: newLeaderUser?.email ?? '',
        },
      }
      onUpdated(updatedGroup)
      setMode('view')
  - catch: setCardError(errorMessage(error))
  - finally: setBusy(false)

  handleDelete():
  - setBusy(true), setCardError(null)
  - try:
      await request<null>('DELETE', API_ENDPOINTS.adminGroup(group.id))
      onDeleted(group.id)
  - catch: setCardError(errorMessage(error)), setMode('view')
  - finally: setBusy(false)

  handleToggleMembers():
  - if expanded → setExpanded(false), return
  - setExpanded(true)
  - if members !== null → return（已加载过，不重复请求）
  - setMembersLoading(true), setMembersError(null)
  - try:
      const data = await request<GroupMemberResponse[]>('GET',
        API_ENDPOINTS.adminGroupMembers(group.id))
      setMembers(data)
  - catch: setMembersError(errorMessage(error))
  - finally: setMembersLoading(false)

  handleMemberAdded(member: GroupMemberResponse):
  - setMembers(prev => prev ? [...prev, member] : [member])
  - onUpdated({ ...group, member_count: group.member_count + 1 })

  handleMemberRemoved(userId: string):
  - setMembers(prev => prev ? prev.filter(m => m.user_id !== userId) : prev)
  - onUpdated({ ...group, member_count: group.member_count - 1 })

  渲染:
  <div className="ag-card">
    {/* ── 头部行 ── */}
    <div className="ag-card-header">
      <div className="ag-card-icon" style={{ background: 'var(--purple)' }}>
        {firstChar(group.name)}
      </div>

      {mode === 'renaming' ? (
        <div className="ag-inline-edit">
          <input className="ag-inline-input" value={editName}
                 onChange={e => setEditName(e.target.value)}
                 autoFocus maxLength={50}
                 onKeyDown={e => { if (e.key === 'Enter') handleRename() }} />
          <button className="admin-btn-secondary admin-btn-sm" type="button"
                  disabled={busy} onClick={handleRename}>
            {busy ? '保存中...' : '确认'}
          </button>
          <button className="admin-btn-secondary admin-btn-sm" type="button"
                  onClick={cancelAction}>取消</button>
        </div>
      ) : (
        <span className="ag-card-name">{group.name}</span>
      )}

      <div className="ag-card-meta">
        <span>组长: <strong>{group.leader.display_name}</strong></span>
        <span>成员: {group.member_count}</span>
        <span>{formatDate(group.created_at)}</span>
      </div>
    </div>

    {/* ── 操作行 ── */}
    <div className="ag-card-actions">
      {mode === 'view' && (
        <>
          <button className="admin-btn-secondary admin-btn-sm" type="button"
                  onClick={startRename}>重命名</button>
          <button className="admin-btn-secondary admin-btn-sm" type="button"
                  onClick={startChangeLeader}>换组长</button>
          <button className="admin-btn-danger admin-btn-sm" type="button"
                  onClick={startDelete}>删除</button>
        </>
      )}

      {mode === 'changing-leader' && (
        <div className="ag-inline-edit">
          <select className="ag-inline-select" value={editLeaderId}
                  onChange={e => setEditLeaderId(e.target.value)}>
            <option value="" disabled>选择新组长</option>
            {allUsers.map(u => (
              <option key={u.id} value={u.id}>
                {u.display_name} ({u.email})
              </option>
            ))}
          </select>
          <button className="admin-btn-secondary admin-btn-sm" type="button"
                  disabled={busy || !editLeaderId} onClick={handleChangeLeader}>
            {busy ? '保存中...' : '确认'}
          </button>
          <button className="admin-btn-secondary admin-btn-sm" type="button"
                  onClick={cancelAction}>取消</button>
        </div>
      )}

      {mode === 'confirming-delete' && (
        <div className="ag-inline-edit">
          <span className="ag-delete-warn">确定删除「{group.name}」？</span>
          <button className="admin-btn-danger admin-btn-sm" type="button"
                  disabled={busy} onClick={handleDelete}>
            {busy ? '删除中...' : '确认删除'}
          </button>
          <button className="admin-btn-secondary admin-btn-sm" type="button"
                  onClick={cancelAction}>取消</button>
        </div>
      )}
    </div>

    {/* ── 卡片级错误 ── */}
    {cardError && <div className="ag-card-error">{cardError}</div>}

    {/* ── 成员区 ── */}
    <MembersSection
      groupId={group.id}
      leaderId={group.leader.id}
      memberCount={group.member_count}
      allUsers={allUsers}
      members={members}
      expanded={expanded}
      membersLoading={membersLoading}
      membersError={membersError}
      onToggle={handleToggleMembers}
      onMemberAdded={handleMemberAdded}
      onMemberRemoved={handleMemberRemoved}
    />
  </div>
  ```

  #### 子组件 `MembersSection`

  ```
  Props:
  interface MembersSectionProps {
    groupId: string
    leaderId: string
    memberCount: number
    allUsers: UserResponse[]
    members: GroupMemberResponse[] | null
    expanded: boolean
    membersLoading: boolean
    membersError: string | null
    onToggle: () => void
    onMemberAdded: (member: GroupMemberResponse) => void
    onMemberRemoved: (userId: string) => void
  }

  状态:
  - adding: boolean                  — 是否正在显示添加行
  - addUserId: string                — 添加行中选中的用户 ID
  - addBusy: boolean                 — 添加请求中
  - addError: string | null          — 添加错误
  - removeBusy: Set<string>          — 正在移除中的成员 user_id 集合
  - removeError: Record<string, string>  — 成员行级移除错误

  handleAddMember():
  - 校验: addUserId 非空，否则 return
  - setAddBusy(true), setAddError(null)
  - try:
      await request<null>('POST', API_ENDPOINTS.adminGroupMembers(groupId),
        { user_id: addUserId } satisfies AddMemberRequest)
      onMemberAdded({
        group_id: groupId,
        user_id: addUserId,
        joined_at: new Date().toISOString(),
      })
      setAdding(false)
      setAddUserId('')
  - catch: setAddError(errorMessage(error))
  - finally: setAddBusy(false)

  handleRemoveMember(userId: string):
  - setRemoveBusy(prev => new Set(prev).add(userId))
  - setRemoveError(prev => { const next = { ...prev }; delete next[userId]; return next })
  - try:
      await request<null>('DELETE', API_ENDPOINTS.adminGroupMember(groupId, userId))
      onMemberRemoved(userId)
  - catch: setRemoveError(prev => ({ ...prev, [userId]: errorMessage(error) }))
  - finally: setRemoveBusy(prev => { const next = new Set(prev); next.delete(userId); return next })

  // 计算可添加的用户列表（排除已是成员的）
  const memberUserIds = new Set(members?.map(m => m.user_id) ?? [])
  const addableUsers = allUsers.filter(u => !memberUserIds.has(u.id))

  渲染:
  <div className="ag-members-section">
    <button className="ag-members-toggle" type="button" onClick={onToggle}>
      {expanded ? '▼' : '▶'} 成员列表 ({memberCount})
    </button>

    {expanded && (
      <div className="ag-members-content">
        {/* 加载中 */}
        {membersLoading && <div className="ag-members-hint">加载中...</div>}

        {/* 加载错误 */}
        {membersError && <div className="ag-members-hint ag-members-hint--error">{membersError}</div>}

        {/* 成员列表 */}
        {members && members.length === 0 && !membersLoading && (
          <div className="ag-members-hint">暂无成员</div>
        )}

        {members && members.length > 0 && (
          <div className="ag-members-list">
            {members.map(member => {
              const user = allUsers.find(u => u.id === member.user_id)
              const isLeader = member.user_id === leaderId
              return (
                <div className="ag-member-row" key={member.user_id}>
                  <div className="ag-member-avatar"
                       style={{ background: avatarColor(member.user_id) }}>
                    {firstChar(user?.display_name ?? '?')}
                  </div>
                  <span className="ag-member-name">
                    {user?.display_name ?? member.user_id.slice(0, 8)}
                    {isLeader && <span className="ag-leader-tag"> (组长)</span>}
                  </span>
                  <span className="ag-member-email">{user?.email ?? ''}</span>
                  <div className="ag-member-actions">
                    {!isLeader && (
                      <button className="ag-member-remove" type="button"
                              disabled={removeBusy.has(member.user_id)}
                              onClick={() => handleRemoveMember(member.user_id)}>
                        {removeBusy.has(member.user_id) ? '...' : '移除'}
                      </button>
                    )}
                  </div>
                  {removeError[member.user_id] && (
                    <span className="row-feedback row-feedback--error">
                      {removeError[member.user_id]}
                    </span>
                  )}
                </div>
              )
            })}
          </div>
        )}

        {/* 添加成员 */}
        {members && !adding && (
          <button className="ag-add-member-btn" type="button"
                  onClick={() => { setAdding(true); setAddUserId(''); setAddError(null) }}>
            + 添加成员
          </button>
        )}

        {adding && (
          <div className="ag-add-member-row">
            <select className="ag-add-member-select" value={addUserId}
                    onChange={e => setAddUserId(e.target.value)}>
              <option value="" disabled>选择用户</option>
              {addableUsers.map(u => (
                <option key={u.id} value={u.id}>
                  {u.display_name} ({u.email})
                </option>
              ))}
            </select>
            <button className="admin-btn-secondary admin-btn-sm" type="button"
                    disabled={addBusy || !addUserId} onClick={handleAddMember}>
              {addBusy ? '添加中...' : '添加'}
            </button>
            <button className="admin-btn-secondary admin-btn-sm" type="button"
                    onClick={() => setAdding(false)}>取消</button>
            {addError && <span className="row-feedback row-feedback--error">{addError}</span>}
          </div>
        )}
      </div>
    )}
  </div>
  ```

  ```css
  /* ═══════════════════════════════════════════════
  ```
