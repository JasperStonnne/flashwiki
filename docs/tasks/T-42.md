  ### 数据库表结构（`backend/migrations/0004_docs_snapshots.up.sql`）

  ```sql
  -- 文档压缩后的完整状态
  CREATE TABLE doc_states (
    node_id            uuid        PRIMARY KEY REFERENCES nodes(id) ON DELETE CASCADE,
    ydoc_state         bytea       NOT NULL,          -- 完整 Y.Doc 二进制
    version            bigint      NOT NULL DEFAULT 0, -- compact 计数器
    markdown_plain     text        NOT NULL DEFAULT '', -- compact 时抽取的纯文本（给搜索用）
    markdown_plain_tsv tsvector,                        -- trigger 自动维护，Go 侧不读写
    last_compacted_at  timestamptz NOT NULL DEFAULT now()
  );

  -- 尚未 compact 的增量 update
  CREATE TABLE doc_updates (
    id         bigserial   PRIMARY KEY,                -- 单调递增，compact 用 max(id) 作为清理界限
    node_id    uuid        NOT NULL REFERENCES nodes(id) ON DELETE CASCADE,
    update     bytea       NOT NULL,                   -- Y.js 单条 binary update
    client_id  text        NOT NULL,                   -- 发送方的 WS client ID
    created_at timestamptz NOT NULL DEFAULT now()
  );
  CREATE INDEX doc_updates_node_idx ON doc_updates(node_id, id);
  ```

  ### 现有 Repo 编码规范（必须严格遵循）

  以 `repository/user_repo.go` 和 `repository/node_permission_repo.go` 为准：

  1. **接口定义**在文件顶部，包含 godoc 注释
  2. **私有结构体** `xxxRepo struct { pool *pgxpool.Pool }`
  3. **构造函数** `NewXxxRepo(pool *pgxpool.Pool) XxxRepo`，返回接口类型
  4. **SQL 用 `const query` 多行字符串**，首行空行 + 缩进对齐
  5. **错误包装** `fmt.Errorf("failed to <英文动作描述>: %w", err)`
  6. **ErrNoRows 判断**：查询类 → `return nil, nil`；写入类 → `return fmt.Errorf("failed to ...: %w", pgx.ErrNoRows)`
  7. **事务方法**接收 `pgx.Tx` 参数，用 `tx.QueryRow` / `tx.Exec` 而非 `r.pool`
     参照 `NodePermissionRepo.ReplacePermissions(ctx, tx, ...)` 和 `DeleteByGroupID(ctx, tx, ...)`
  8. **slice 初始化** `make([]T, 0)` 不用 `var x []T`

  完整文件内容：

  ```go
  package models

  import (
  	"time"

  	"github.com/google/uuid"
  )

  // DocState 存储文档的最近一次 compact 后的完整 Y.Doc 二进制状态。
  // 表: doc_states, PK: node_id
  type DocState struct {
  	NodeID          uuid.UUID `db:"node_id"`
  	YDocState       []byte    `db:"ydoc_state"`
  	Version         int64     `db:"version"`
  	MarkdownPlain   string    `db:"markdown_plain"`
  	LastCompactedAt time.Time `db:"last_compacted_at"`
  }

  // DocUpdate 存储单条 Y.js 增量更新（尚未 compact 的）。
  // 表: doc_updates, PK: id (bigserial)
  type DocUpdate struct {
  	ID        int64     `db:"id"`
  	NodeID    uuid.UUID `db:"node_id"`
  	Update    []byte    `db:"update"`
  	ClientID  string    `db:"client_id"`
  	CreatedAt time.Time `db:"created_at"`
  }
  ```

  字段要点：
  - `DocState` 不包含 `markdown_plain_tsv` — 由 PostgreSQL trigger 自动维护，Go 侧不读写
  - `DocUpdate.Update` 字段名大写 `Update`，db tag `"update"`
  - `DocUpdate.ID` 类型 `int64` 对应 `bigserial`
  - `DocState.YDocState` 类型 `[]byte` 对应 `bytea`
  - 所有 time 用 `time.Time`（UTC）
  - 所有 uuid 用 `uuid.UUID`

  ### 接口定义

  ```go
  type DocStateRepo interface {
  	// GetOrInit 获取文档状态；如果不存在则插入空 Y.Doc 初始状态并返回。
  	// 空 Y.Doc = []byte{0, 0}。幂等：多次调用同一 nodeID 不报错。
  	// 不走事务，直接用 pool。
  	GetOrInit(ctx context.Context, nodeID uuid.UUID) (*models.DocState, error)

  	// GetForUpdate 在事务内获取 doc_states 行并加行锁。
  	// 用于 compact 流程，必须在 pgx.Tx 上下文中调用。
  	// 如果行不存在返回 (nil, nil)。
  	GetForUpdate(ctx context.Context, tx pgx.Tx, nodeID uuid.UUID) (*models.DocState, error)

  	// UpdateAfterCompact 在事务内更新合并后的文档状态。
  	// 设置 ydoc_state、markdown_plain、version+1、last_compacted_at=now()。
  	UpdateAfterCompact(ctx context.Context, tx pgx.Tx, nodeID uuid.UUID, ydocState []byte, markdownPlain string) error
  }
  ```

  ```go

  func NewDocStateRepo(pool *pgxpool.Pool) DocStateRepo {
  ```

  ### 方法 1: GetOrInit

  ```go
  func (r *docStateRepo) GetOrInit(ctx context.Context, nodeID uuid.UUID) (*models.DocState, error) {
  	// 空 Y.Doc 最小二进制表示：
  	// 字节 0 = 0 个 client structs
  	// 字节 1 = 0 个 delete set entries
  	emptyYDoc := []byte{0, 0}

  	// 幂等插入：已存在则跳过
  	const insertQuery = `
  INSERT INTO doc_states (node_id, ydoc_state)
  VALUES ($1, $2)
  ON CONFLICT (node_id) DO NOTHING
  `
  	if _, err := r.pool.Exec(ctx, insertQuery, nodeID, emptyYDoc); err != nil {
  		return nil, fmt.Errorf("failed to init doc state: %w", err)
  	}

  	// 读取（无论是刚插入的还是已有的）
  	const selectQuery = `
  SELECT node_id, ydoc_state, version, markdown_plain, last_compacted_at
  FROM doc_states
  WHERE node_id = $1
  `
  	ds := &models.DocState{}
  	if err := r.pool.QueryRow(ctx, selectQuery, nodeID).Scan(
  		&ds.NodeID,
  		&ds.YDocState,
  		&ds.Version,
  		&ds.MarkdownPlain,
  		&ds.LastCompactedAt,
  	); err != nil {
  		return nil, fmt.Errorf("failed to get doc state: %w", err)
  	}

  	return ds, nil
  }
  ```

  关键点：
  - 不用 `RETURNING` — `ON CONFLICT DO NOTHING` 时不返回行
  - 分两步：INSERT 幂等 + SELECT 查询
  - `emptyYDoc` = `[]byte{0, 0}`，不是 `[]byte{}`（空 slice）也不是 `nil`
  - 用 `r.pool.Exec` 和 `r.pool.QueryRow`（不走事务）
  - SELECT 不需要 ErrNoRows 判断 — INSERT 保证了行一定存在

  ### 方法 2: GetForUpdate

  ```go
  func (r *docStateRepo) GetForUpdate(ctx context.Context, tx pgx.Tx, nodeID uuid.UUID) (*models.DocState, error) {
  	const query = `
  SELECT node_id, ydoc_state, version, markdown_plain, last_compacted_at
  FROM doc_states
  WHERE node_id = $1
  FOR UPDATE
  `
  	ds := &models.DocState{}
  	if err := tx.QueryRow(ctx, query, nodeID).Scan(
  		&ds.NodeID,
  		&ds.YDocState,
  		&ds.Version,
  		&ds.MarkdownPlain,
  		&ds.LastCompactedAt,
  	); err != nil {
  		if errors.Is(err, pgx.ErrNoRows) {
  			return nil, nil
  		}
  		return nil, fmt.Errorf("failed to get doc state for update: %w", err)
  	}

  	return ds, nil
  }
  ```

  关键点：
  - 用 `tx.QueryRow` 而不是 `r.pool.QueryRow` — 必须走事务连接才能持有行锁
  - `FOR UPDATE` 加行锁，防止并发 compact 冲突
  - ErrNoRows → `return nil, nil`（与 `NodeRepo.FindByID` 模式一致）
  - Scan 字段顺序与 SELECT 列顺序严格对齐

  ### 方法 3: UpdateAfterCompact

  ```go
  func (r *docStateRepo) UpdateAfterCompact(ctx context.Context, tx pgx.Tx, nodeID uuid.UUID, ydocState []byte, markdownPlain string) error {
  	const query = `
  UPDATE doc_states
  SET ydoc_state = $2,
      markdown_plain = $3,
      version = version + 1,
      last_compacted_at = now()
  WHERE node_id = $1
  `
  	tag, err := tx.Exec(ctx, query, nodeID, ydocState, markdownPlain)
  	if err != nil {
  		return fmt.Errorf("failed to update doc state after compact: %w", err)
  	}
  	if tag.RowsAffected() == 0 {
  		return fmt.Errorf("failed to update doc state after compact: %w", pgx.ErrNoRows)
  	}

  	return nil
  }
  ```

  关键点：
  - 用 `tx.Exec` 而不是 `r.pool.Exec` — 必须走事务
  - `version = version + 1` 在 SQL 侧自增，不依赖 Go 传值
  - `last_compacted_at = now()` 在 SQL 侧取时间
  - `markdown_plain` 更新会自动触发 `doc_states_tsv_trigger` 更新 tsvector
  - RowsAffected == 0 → 返回 ErrNoRows 包装错误（与 `UserRepo.UpdateRole` 模式一致）

  ### 接口定义

  ```go
  type DocUpdateRepo interface {
  	// Append 追加一条增量 Y.js update 到 doc_updates 表。
  	// 由 Hub.handleUpdate 调用（每条 WS update 消息一次）。
  	// 不返回插入的 id。
  	Append(ctx context.Context, nodeID uuid.UUID, update []byte, clientID string) error

  	// ListByNode 获取某节点的所有 pending updates，按 id ASC 排序。
  	// 用途 1: sync_step1 发送给新连接的客户端（全量 pending）。
  	// 用途 2: compact 合并前读取所有待合并 updates。
  	ListByNode(ctx context.Context, nodeID uuid.UUID) ([]models.DocUpdate, error)

  	// CountByNode 获取某节点的 pending update 数量。
  	// 用于 compact 候选判断（count ≥ 200 触发）。
  	CountByNode(ctx context.Context, nodeID uuid.UUID) (int64, error)

  	// DeleteUpTo 在事务内删除 node_id 匹配且 id ≤ maxID 的所有 update。
  	// compact 合并成功后清理已合并的行。
  	DeleteUpTo(ctx context.Context, tx pgx.Tx, nodeID uuid.UUID, maxID int64) error

  	// FindCompactCandidates 查找需要 compact 的 node_id 列表。
  	// 满足任一条件即命中:
  	//   条件 A: 该 node 的 doc_updates 行数 ≥ countThreshold
  	//   条件 B: doc_states.last_compacted_at < now()-staleInterval 且存在 pending updates
  	// staleInterval 格式: PostgreSQL interval 字符串，如 "30 minutes"。
  	FindCompactCandidates(ctx context.Context, countThreshold int64, staleInterval string) ([]uuid.UUID, error)
  }
  ```

  ### 方法 1: Append

  ```go
  func (r *docUpdateRepo) Append(ctx context.Context, nodeID uuid.UUID, update []byte, clientID string) error {
  	const query = `
  INSERT INTO doc_updates (node_id, update, client_id)
  VALUES ($1, $2, $3)
  `
  	if _, err := r.pool.Exec(ctx, query, nodeID, update, clientID); err != nil {
  		return fmt.Errorf("failed to append doc update: %w", err)
  	}

  	return nil
  }
  ```

  关键点：
  - SQL 列名 `update` 是 PostgreSQL 非保留关键字，在 INSERT 列列表中可直接使用
  - 如果 pgx 报关键字冲突，改为 `"update"`（双引号）— 但正常不会
  - 不返回 id（INSERT 不加 RETURNING）
  - 不走事务（Hub 中异步 goroutine 调用，独立连接即可）

  ### 方法 2: ListByNode

  ```go
  func (r *docUpdateRepo) ListByNode(ctx context.Context, nodeID uuid.UUID) ([]models.DocUpdate, error) {
  	const query = `
  SELECT id, node_id, update, client_id, created_at
  FROM doc_updates
  WHERE node_id = $1
  ORDER BY id ASC
  `
  	rows, err := r.pool.Query(ctx, query, nodeID)
  	if err != nil {
  		return nil, fmt.Errorf("failed to list doc updates: %w", err)
  	}
  	defer rows.Close()

  	updates := make([]models.DocUpdate, 0)
  	for rows.Next() {
  		var u models.DocUpdate
  		if err := rows.Scan(
  			&u.ID,
  			&u.NodeID,
  			&u.Update,
  			&u.ClientID,
  			&u.CreatedAt,
  		); err != nil {
  			return nil, fmt.Errorf("failed to list doc updates: %w", err)
  		}
  		updates = append(updates, u)
  	}
  	if err := rows.Err(); err != nil {
  		return nil, fmt.Errorf("failed to list doc updates: %w", err)
  	}

  	return updates, nil
  }
  ```

  关键点：
  - 返回 `[]models.DocUpdate`（值类型 slice）不是 `[]*models.DocUpdate`
  - Scan 每行用 `var u models.DocUpdate`（栈变量），append 时拷贝进 slice
  - `ORDER BY id ASC` — 保证客户端按生成顺序应用 updates
  - 空结果返回 `[]models.DocUpdate{}`（make 初始化），不是 nil
  - `defer rows.Close()` 紧跟 Query 成功后
  - 循环后检查 `rows.Err()`

  ### 方法 4: DeleteUpTo

  ```go
  func (r *docUpdateRepo) DeleteUpTo(ctx context.Context, tx pgx.Tx, nodeID uuid.UUID, maxID int64) error {
  	const query = `
  DELETE FROM doc_updates
  WHERE node_id = $1 AND id <= $2
  `
  	if _, err := tx.Exec(ctx, query, nodeID, maxID); err != nil {
  		return fmt.Errorf("failed to delete doc updates up to id %d: %w", maxID, err)
  	}

  	return nil
  }
  ```

  关键点：
  - 用 `tx.Exec` 而不是 `r.pool.Exec` — compact 事务内执行
  - 不检查 RowsAffected — 删除 0 行也合法（可能在 ListByNode 和 DeleteUpTo 之间没有新 update）
  - 错误消息包含 `maxID` 值，便于排查

  ### 方法 5: FindCompactCandidates

  ```go
  func (r *docUpdateRepo) FindCompactCandidates(ctx context.Context, countThreshold int64, staleInterval string) ([]uuid.UUID, error) {
  	// staleInterval 是固定值（如 "30 minutes"），由调用方传入，不是用户输入。
  	// PostgreSQL 不支持 $n 参数化 interval 字面量，因此用 fmt.Sprintf 拼接。
  	query := fmt.Sprintf(`
  SELECT node_id FROM doc_updates
  GROUP BY node_id
  HAVING count(*) >= $1

  UNION

  SELECT ds.node_id FROM doc_states ds
  WHERE ds.last_compacted_at < now() - interval '%s'
    AND EXISTS (SELECT 1 FROM doc_updates du WHERE du.node_id = ds.node_id)
  `, staleInterval)

  	rows, err := r.pool.Query(ctx, query, countThreshold)
  	if err != nil {
  		return nil, fmt.Errorf("failed to find compact candidates: %w", err)
  	}
  	defer rows.Close()

  	ids := make([]uuid.UUID, 0)
  	for rows.Next() {
  		var id uuid.UUID
  		if err := rows.Scan(&id); err != nil {
  			return nil, fmt.Errorf("failed to find compact candidates: %w", err)
  		}
  		ids = append(ids, id)
  	}
  	if err := rows.Err(); err != nil {
  		return nil, fmt.Errorf("failed to find compact candidates: %w", err)
  	}

  	return ids, nil
  }
  ```

  关键点：
  - `fmt.Sprintf` 拼接 interval — 安全：`staleInterval` 由 Go 代码传入固定字符串 `"30 minutes"`，不是用户输入
  - `$1` 参数化 `countThreshold` — 安全
  - `UNION` 自动去重（同一 node_id 可能同时满足两个条件）
  - 条件 A: `count(*) >= $1` — 对应 D-09a 阈值 200
  - 条件 B: `last_compacted_at < now() - interval '30 minutes'` + `EXISTS pending` — 对应 D-09a 时间阈值
  - 空结果返回 `[]uuid.UUID{}`（make 初始化），调用方遍历时自然跳过
